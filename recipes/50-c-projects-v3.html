01
Calculator (CLI) Console Beginner
A command-line calculator supporting +, −, ×, ÷, modulo, and power. Runs in a loop so users can keep calculating without restarting. Handles division by zero and invalid input gracefully.
Compile: gcc calculator.c -o calculator -lm

📚 Mini Course — C Basics & I/O

1️⃣ Variables and Data Types
In C, every variable must be declared with a type. Use double for decimal numbers (more precision than float). Use char for single characters like the operator (+,-,*,/).

Example: double a, b, result; and char op;
2️⃣ Reading Input with scanf()
scanf() reads formatted input from the terminal. The format string describes what to expect. Use %lf for double, %c for char, %d for int.

Example: scanf("%lf %c %lf", &a, &op, &b); — the & passes the address of the variable so scanf can write to it.
3️⃣ Switch Statement
Use switch(op) to branch on the operator character. Each case handles one operator. Always add break to prevent fall-through. Add a default case for unknown operators.
4️⃣ Loops with while(1)
while(1) creates an infinite loop. Use break to exit it when the user types 'q'. This pattern is very common in C menu-driven programs.
5️⃣ The math.h Library
Include #include <math.h> to use pow(base, exp) for exponentiation. You must link it at compile time: gcc calculator.c -o calc -lm (the -lm flag links the math library).

🗺 Build Plan

  1. Set up the project: create calculator.c, include stdio.h and math.h
  2. Declare variables for a, b, op, and result
  3. Write a while(1) loop with a prompt
  4. Use scanf to read a op b format
  5. Add a switch on op with cases for +, -, *, /, %, ^
  6. Handle division by zero in the '/' and '%' cases
  7. Print the result with printf
  8. Let user quit by pressing 'q'

💡 Hints

1
Use scanf(" %lf %c %lf", &a, &op, &b) — notice the space before %c! It skips whitespace/newlines left in the buffer from previous input. Without it, op may capture a newline.
2
To handle the 'q' quit case, check if scanf returns 3 (it returns the number of items successfully read). If it doesn't return 3, read a single char and check if it's 'q'.
3
For modulo with doubles, use fmod(a, b) from math.h, or cast to int: (int)a % (int)b. For the power operator '^', use pow(a, b).
🔓 Solution — calculator.c
#include <stdio.h>
#include <math.h>

int main() {
    double a, b, result;
    char op;

    printf("╔══════════════════════════╗\n");
    printf("║     C CLI Calculator     ║\n");
    printf("║  Operators: + - * / %% ^ ║\n");
    printf("║  Type 'q' to quit        ║\n");
    printf("╚══════════════════════════╝\n\n");

    while (1) {
        printf("› ");
        /* Try to read: number operator number */
        if (scanf(" %lf %c %lf", &a, &op, &b) != 3) {
            /* Didn't get 3 values — check for 'q' */
            char c;
            scanf(" %c", &c);
            if (c == 'q' || c == 'Q') {
                printf("Goodbye!\n");
                break;
            }
            /* Clear input buffer */
            while (getchar() != '\n');
            printf("Invalid input. Try: 5 + 3\n\n");
            continue;
        }

        switch (op) {
            case '+': result = a + b; break;
            case '-': result = a - b; break;
            case '*': result = a * b; break;
            case '/':
                if (b == 0.0) { printf("Error: division by zero!\n\n"); continue; }
                result = a / b;
                break;
            case '%':
                if ((int)b == 0) { printf("Error: modulo by zero!\n\n"); continue; }
                result = fmod(a, b);
                break;
            case '^':
                result = pow(a, b);
                break;
            default:
                printf("Unknown operator '%c'. Use + - * / %% ^\n\n", op);
                continue;
        }

        printf("  = %.10g\n\n", result);
    }

    return 0;
}

Unit Converter Console Beginner

📚 Mini Course — Functions & Menus in C

1️⃣ Functions in C
Functions let you break code into reusable blocks. Declare a function before main() or use a prototype. A function that converts km to miles: double km_to_miles(double km) { return km * 0.621371; }
2️⃣ Nested Switch / Menu Pattern
A common C pattern: outer switch picks the category (Length, Weight, Temperature), inner switch picks the direction. Use printf to display the menu and scanf("%d", &choice) to read integer selections.
3️⃣ Constants with #define
Use #define KM_TO_MI 0.621371 for conversion factors. These are preprocessor macros — replaced by the compiler before compilation. They make your code more readable and easier to update.

🗺 Build Plan

  1. Define conversion constants with #define
  2. Write a conversion function for each unit pair
  3. Create a print_menu() function
  4. Main loop: display menu → read choice → display subcategory menu → read value → print result
  5. Add input validation (reject negative temperatures in Kelvin, etc.)

💡 Hints

1
Temperature conversions need a formula, not just a multiplier. Celsius to Fahrenheit: (c * 9.0/5.0) + 32. Notice 9.0/5.0 not 9/5 — integer division in C would give 1!
2
After scanf("%d", &choice), the newline stays in the buffer. If you later read a string or char, add while(getchar() != '\n'); to flush it.
3
Use a do { ... } while(choice != 0); loop for "press 0 to go back" menus. It always runs at least once, which is perfect for menu loops.

#include <stdio.h>

/* ── Conversion factors ── */
#define KM_TO_MI    0.621371
#define KG_TO_LBS   2.20462
#define L_TO_GAL    0.264172
#define IN_TO_CM    2.54
#define M_TO_FT     3.28084

void clear_buffer() { while (getchar() != '\n'); }

/* ── Length ── */
void length_menu() {
    int ch; double val;
    printf("\n  Length: 1) km→mi  2) mi→km  3) m→ft  4) ft→m  5) in→cm  6) cm→in\n  Choice: ");
    scanf("%d", &ch); clear_buffer();
    printf("  Enter value: "); scanf("%lf", &val); clear_buffer();
    switch(ch) {
        case 1: printf("  %.4f km = %.4f miles\n", val, val*KM_TO_MI); break;
        case 2: printf("  %.4f miles = %.4f km\n", val, val/KM_TO_MI); break;
        case 3: printf("  %.4f m = %.4f ft\n", val, val*M_TO_FT); break;
        case 4: printf("  %.4f ft = %.4f m\n", val, val/M_TO_FT); break;
        case 5: printf("  %.4f in = %.4f cm\n", val, val*IN_TO_CM); break;
        case 6: printf("  %.4f cm = %.4f in\n", val, val/IN_TO_CM); break;
        default: printf("  Invalid choice.\n");
    }
}

/* ── Weight ── */
void weight_menu() {

}

/* ── Temperature ── */

        case 3: printf("  %.2f°C = %.2f K\n", val, val+273.15); break;
        case 4: printf("  %.2f K = %.2f°C\n", val, val-273.15); break;

}

/* ── Volume ── */
void volume_menu() {
    int ch; double val;

    scanf("%d", &ch); clear_buffer();
    printf("  Enter value: "); scanf("%lf", &val); clear_buffer();
    switch(ch) {
        case 1: printf("  %.4f L = %.4f gallons\n", val, val*L_TO_GAL); break;
        case 2: printf("  %.4f gallons = %.4f L\n", val, val/L_TO_GAL); break;

}

    int choice;
    printf("╔══════════════════════════╗\n");
    printf("║     Unit Converter       ║\n");
    printf("╚══════════════════════════╝\n");

        printf("\n  1) Length\n  2) Weight\n  3) Temperature\n  4) Volume\n  0) Exit\n  › ");
        scanf("%d", &choice); clear_buffer();

            case 1: length_menu(); break;
            case 2: weight_menu(); break;
            case 3: temp_menu(); break;
            case 4: volume_menu(); break;

        }
    } while (choice != 0);

    return 0;
}

Number Guessing Game Console Beginner
The program picks a random number 1–100. The user guesses and receives "too high / too low" hints. Tracks attempts and shows a high-score (fewest guesses).
Compile: gcc guessing_game.c -o guess

📚 Mini Course — Random Numbers & Game Loops

1️⃣ Generating Random Numbers
Use rand() from stdlib.h. Without seeding, it returns the same sequence every run. Seed it with the current time: srand(time(NULL));. To get a number 1–100: int secret = (rand() % 100) + 1;
2️⃣ Comparison and Branching
Use if / else if / else to compare the guess with the secret number. if (guess > secret) → "Too high", if (guess < secret) → "Too low", else → "Correct!"
3️⃣ Counting Attempts
Declare int attempts = 0; and increment it with attempts++; inside the guess loop. Compare it against a best_score variable to track the high score across multiple game rounds.

🗺 Build Plan

  1. Include stdio.h, stdlib.h, time.h
  2. Seed srand(time(NULL)) once in main()
  3. Wrap the game in an outer do-while (play again?)
  4. Inner while loop: read guess, compare, increment counter
  5. On win: show attempts, check against best_score, offer replay

💡 Hints

1
rand() % 100 gives 0–99. Add 1 to get 1–100: int secret = (rand() % 100) + 1;
2
Initialize int best = 999; for "no best score yet". After each win, check if (attempts < best) best = attempts;
3
For "play again?", read a char with scanf(" %c", &again). Check again == 'y' || again == 'Y'.
🔓 Solution — guessing_game.c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    srand((unsigned int)time(NULL));

    int best_score = 999;
    char again;

    printf("╔══════════════════════════════╗\n");
    printf("║    Number Guessing Game      ║\n");
    printf("║  Guess a number from 1–100   ║\n");
    printf("╚══════════════════════════════╝\n");

    do {
        int secret = (rand() % 100) + 1;
        int guess, attempts = 0;
        int won = 0;

        while (!won) {
            printf("\nYour guess: ");
            if (scanf("%d", &guess) != 1) {
                while (getchar() != '\n');
                printf("Please enter a number!\n");
                continue;
            }

            attempts++;

            if (guess < 1 || guess > 100) {
                printf("Please guess between 1 and 100.\n");
                attempts--;
                continue;
            }

            if (guess < secret)
                printf("  ↑ Too low!  (attempt %d)\n", attempts);
            else if (guess > secret)
                printf("  ↓ Too high! (attempt %d)\n", attempts);
            else {
                printf("  ✓ Correct! The number was %d.\n", secret);
                printf("  You got it in %d attempt%s!\n", attempts, attempts == 1 ? "" : "s");

                if (attempts < best_score) {
                    best_score = attempts;
                    printf("  🏆 New best score: %d!\n", best_score);
                } else {
                    printf("  Best score: %d\n", best_score);
                }
                won = 1;
            }
        }

        printf("\nPlay again? (y/n): ");
        scanf(" %c", &again);

    } while (again == 'y' || again == 'Y');

    printf("\nThanks for playing! Best score: %d\n", best_score == 999 ? 0 : best_score);
    return 0;
}

Fibonacci Generator Console Beginner
Generate Fibonacci sequences using iterative, recursive, and memoized approaches. Compare their performance using clock timing to illustrate why memoization matters.
Compile: gcc fibonacci.c -o fibonacci

📚 Mini Course — Recursion, Arrays & Timing

1️⃣ Recursion
A recursive function calls itself. For Fibonacci: F(n) = F(n-1) + F(n-2) with base cases F(0) = 0 and F(1) = 1. Every recursive function needs base cases to stop the recursion, otherwise you get infinite recursion and a stack overflow.
2️⃣ Iterative vs Recursive
Recursive Fibonacci has exponential time complexity O(2^n) — it recalculates the same values millions of times for large n. Iterative Fibonacci uses O(n) time and O(1) space by just keeping two previous values and updating them in a loop.
3️⃣ Memoization (Top-Down DP)
Memoization stores results of expensive function calls in an array. Before computing F(n), check if memo[n] != -1. If it's cached, return it directly. This brings recursive Fibonacci down to O(n) time.
4️⃣ Timing with clock()
Use clock_t start = clock(); before and clock_t end = clock(); after a block of code. Calculate elapsed time: double time = (double)(end - start) / CLOCKS_PER_SEC;

🗺 Build Plan

  1. Write the iterative version first (simplest)
  2. Write the naive recursive version
  3. Write the memoized recursive version with a long long memo[100] array
  4. Add clock() timing around each approach
  5. Print all results and compare timing for n=40

💡 Hints

1
Use long long instead of int for Fibonacci values — they grow very fast! F(50) = 12,586,269,025 which overflows a 32-bit int.
2
Initialize your memo array with -1 using: long long memo[100]; memset(memo, -1, sizeof(memo)); from string.h.
3
Keep the naive recursive version for n ≤ 45. For n > 45 it will take seconds. The memoized version handles n=90 instantly.
🔓 Solution — fibonacci.c
#include <stdio.h>
#include <string.h>
#include <time.h>

/* ── Iterative: O(n) time, O(1) space ── */
long long fib_iterative(int n) {
    if (n <= 1) return n;
    long long a = 0, b = 1, c;
    for (int i = 2; i <= n; i++) {
        c = a + b;
        a = b;
        b = c;
    }
    return b;
}

/* ── Naive Recursive: O(2^n) time ── */
long long fib_recursive(int n) {
    if (n <= 1) return n;
    return fib_recursive(n - 1) + fib_recursive(n - 2);
}

/* ── Memoized Recursive: O(n) time ── */
static long long memo[100];

long long fib_memo(int n) {
    if (n <= 1) return n;
    if (memo[n] != -1) return memo[n];
    memo[n] = fib_memo(n - 1) + fib_memo(n - 2);
    return memo[n];
}

int main() {
    int n;
    printf("╔═══════════════════════════════╗\n");
    printf("║    Fibonacci Generator        ║\n");
    printf("╚═══════════════════════════════╝\n");

    printf("Enter N (1-90): ");
    scanf("%d", &n);
    if (n < 1 || n > 90) { printf("Please enter 1-90.\n"); return 1; }

    /* ── Print sequence ── */
    printf("\nFibonacci sequence (iterative):\n");
    for (int i = 0; i <= n; i++) {
        printf("  F(%2d) = %lld\n", i, fib_iterative(i));
    }

    /* ── Benchmark for fixed n ── */
    int bench_n = n < 40 ? n : 40;
    printf("\n── Performance Benchmark (n=%d) ──\n", bench_n);

    /* Iterative timing */
    clock_t s = clock();
    for (int i = 0; i < 100000; i++) fib_iterative(bench_n);
    double t1 = (double)(clock() - s) / CLOCKS_PER_SEC;
    printf("Iterative  : %lld  (%.6fs × 100k runs)\n", fib_iterative(bench_n), t1);

    /* Naive recursive timing (only for small n) */
    if (bench_n <= 35) {
        s = clock();
        long long r = fib_recursive(bench_n);
        double t2 = (double)(clock() - s) / CLOCKS_PER_SEC;
        printf("Recursive  : %lld  (%.6fs single call)\n", r, t2);
    } else {
        printf("Recursive  : (skipped for n>35 — too slow)\n");
    }

    /* Memoized timing */
    memset(memo, -1, sizeof(memo));
    s = clock();
    for (int i = 0; i < 100000; i++) {
        memset(memo, -1, sizeof(memo));
        fib_memo(bench_n);
    }
    double t3 = (double)(clock() - s) / CLOCKS_PER_SEC;
    memset(memo, -1, sizeof(memo));
    printf("Memoized   : %lld  (%.6fs × 100k runs)\n", fib_memo(bench_n), t3);

    return 0;
}

Prime Sieve (Eratosthenes) Console Beginner
Find all primes up to N using the Sieve of Eratosthenes. Display them in a formatted grid and benchmark how long it takes to sieve up to 1,000,000.
Compile: gcc prime_sieve.c -o primes

📚 Mini Course — Arrays, Booleans & Algorithms

1️⃣ Boolean Arrays in C
C doesn't have a built-in boolean until C99. Use #include <stdbool.h> for bool, true, false. Alternatively use int arrays where 0 = false, 1 = true. For the sieve, declare: bool is_prime[MAX+1];
2️⃣ The Sieve Algorithm
1) Mark all numbers 2..N as prime. 2) For each prime p starting at 2: mark all multiples of p (p², p²+p, p²+2p…) as NOT prime. 3) Only need to go up to √N. The key insight: start marking from p² not 2p because smaller multiples were already marked by earlier primes.
3️⃣ memset() for Fast Initialization
memset(is_prime, 1, sizeof(is_prime)); sets every byte to 1 (true) instantly — much faster than a loop for large arrays. Then manually set is_prime[0] = is_prime[1] = false; since 0 and 1 are not prime.

🗺 Build Plan

  1. Declare a bool is_prime[N+1] array
  2. Initialize all to true with memset
  3. Set index 0 and 1 to false
  4. Loop p from 2 to √N: if prime, mark multiples from p² as false
  5. Print all i where is_prime[i] is true
  6. Count primes and show timing

💡 Hints

1
For the inner loop, start at j = p*p, not j = 2*p. All smaller multiples of p were already marked by previous primes. This is the critical optimization.
2
The outer loop only needs to run to sqrt(N). Use: for (int p = 2; (long long)p*p <= n; p++). The cast to long long prevents overflow for large n.
3
For large N (1M+), allocate the array on the heap with malloc to avoid stack overflow: bool *is_prime = malloc((n+1) * sizeof(bool));
🔓 Solution — prime_sieve.c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>
#include <math.h>
#include <time.h>

void sieve(int n) {
    /* Allocate on heap for large N */
    bool *is_prime = (bool *)malloc((n + 1) * sizeof(bool));
    if (!is_prime) { fprintf(stderr, "Out of memory\n"); exit(1); }

    /* Mark all as prime */
    memset(is_prime, 1, (n + 1) * sizeof(bool));
    is_prime[0] = is_prime[1] = false;

    clock_t start = clock();

    /* Sieve */
    for (long long p = 2; p * p <= n; p++) {
        if (is_prime[p]) {
            for (long long j = p * p; j <= n; j += p) {
                is_prime[j] = false;
            }
        }
    }

    double elapsed = (double)(clock() - start) / CLOCKS_PER_SEC;

    /* Count and print */
    int count = 0;
    printf("Primes up to %d:\n", n);

    if (n <= 1000) {
        for (int i = 2; i <= n; i++) {
            if (is_prime[i]) {
                printf("%6d", i);
                count++;
                if (count % 10 == 0) putchar('\n');
            }
        }
        if (count % 10 != 0) putchar('\n');
    } else {
        /* Just count for large N */
        for (int i = 2; i <= n; i++)
            if (is_prime[i]) count++;
        printf("  (too many to display — showing count only)\n");
    }

    printf("\nTotal primes found: %d\n", count);
    printf("Sieve time: %.6f seconds\n", elapsed);

    /* Show first 20 and last 20 for large N */
    if (n > 1000) {
        int shown = 0;
        printf("\nFirst 20 primes: ");
        for (int i = 2; i <= n && shown < 20; i++) {
            if (is_prime[i]) { printf("%d ", i); shown++; }
        }
        shown = 0;
        printf("\nLast 20 primes:  ");
        int last20[20]; int idx = 0;
        for (int i = n; i >= 2 && idx < 20; i--)
            if (is_prime[i]) last20[idx++] = i;
        for (int i = idx-1; i >= 0; i--) printf("%d ", last20[i]);
        printf("\n");
    }

    free(is_prime);
}

int main() {
    int n;
    printf("╔════════════════════════════╗\n");
    printf("║  Sieve of Eratosthenes     ║\n");
    printf("╚════════════════════════════╝\n");
    printf("Find primes up to N: ");
    scanf("%d", &n);

    if (n < 2) { printf("No primes below 2.\n"); return 0; }
    if (n > 10000000) { printf("Max is 10,000,000\n"); return 1; }

    sieve(n);
    return 0;
}

Caesar Cipher Console Beginner
Encrypt and decrypt text using the Caesar cipher. Add a brute-force mode that tries all 25 shifts and displays the most English-like result using letter frequency analysis.
Compile: gcc caesar.c -o caesar

📚 Mini Course — Strings, Characters & Modular Arithmetic

1️⃣ Strings in C
Strings are arrays of char ending with a null terminator '\0'. Declare with: char text[256]; and read with fgets(text, sizeof(text), stdin);. Never use gets() — it's unsafe!
2️⃣ Character Operations
Characters in C are just integers. 'A' = 65, 'Z' = 90, 'a' = 97, 'z' = 122. Use ctype.h functions: isupper(c), islower(c), isalpha(c). To shift: (c - 'A' + shift) % 26 + 'A'
3️⃣ Modular Arithmetic (Wrap-around)
The key formula: ((c - base + shift) % 26 + 26) % 26 + base. The +26 before the second %26 handles negative shifts in C (since C's % can return negative for negative operands).
4️⃣ Frequency Analysis
Score each possible decryption by counting how often the most common English letters (ETAOIN SHRDLU) appear. The shift producing the highest score is most likely the correct one.

🗺 Build Plan

  1. Write encrypt(char *text, int shift) and decrypt() functions
  2. Loop through each character, only shift alphabetic chars
  3. Write a score(char *text) function for frequency analysis
  4. Brute-force: try shifts 1–25, score each, print the best
  5. Add a menu: encrypt / decrypt / brute-force

💡 Hints

1
The Caesar shift formula for uppercase: char shifted = (c - 'A' + shift + 26) % 26 + 'A';. The +26 ensures we never take modulo of a negative number when decrypting.
2
Use fgets() to read text with spaces. It includes the trailing newline — remove it with: text[strcspn(text, "\n")] = '\0';
3
For frequency scoring, count occurrences of each letter a-z, then compute a weighted sum: score += freq['e'-'a'] * 12 + freq['t'-'a'] * 9 + ...
🔓 Solution — caesar.c

};  /* a b c d e f g h i j k l m n o p q r s t u v w x y z */

char shift_char(char c, int shift) {
    if (isalpha(c)) {
        char base = isupper(c) ? 'A' : 'a';
        return (char)((c - base + shift + 26) % 26 + base);
    }
    return c;
}

void caesar(const char *input, char *output, int shift) {
    int len = strlen(input);
    for (int i = 0; i < len; i++)
        output[i] = shift_char(input[i], shift);
    output[len] = '\0';
}

/* Score text by English-likeness */
int score_text(const char *text) {
    int freq[26] = {0};
    int total = 0;
    for (int i = 0; text[i]; i++) {
        if (isalpha(text[i])) {
            freq[tolower(text[i]) - 'a']++;
            total++;
        }
    }
    if (total == 0) return 0;
    int score = 0;
    for (int i = 0; i < 26; i++)
        score += freq[i] * FREQ[i];
    return score;
}

    }

        for (int j = 0; j < A->cols; j++)

}

#include <stdio.h>
#include <string.h>
#include <ctype.h>

static const int    VALUES[]   = {1000,900,500,400,100,90,50,40,10,9,5,4,1};
static const char  *SYMBOLS[]  = {"M","CM","D","CD","C","XC","L","XL","X","IX","V","IV","I"};
static const int    N_SYMBOLS  = 13;

void int_to_roman(int n, char *out) {
    out[0] = '\0';
    if (n < 1 || n > 3999) { strcpy(out, "OUT OF RANGE"); return; }
    for (int i = 0; i < N_SYMBOLS; i++) {
        while (n >= VALUES[i]) {
            strcat(out, SYMBOLS[i]);
            n -= VALUES[i];
        }
    }
}

int roman_val(char c) {
    switch(toupper(c)) {
        case 'I': return 1;    case 'V': return 5;
        case 'X': return 10;   case 'L': return 50;
        case 'C': return 100;  case 'D': return 500;
        case 'M': return 1000;
        default:  return -1;
    }
}

int roman_to_int(const char *s) {
    int total = 0;
    int len = strlen(s);
    for (int i = 0; i < len; i++) {
        int curr = roman_val(s[i]);
        if (curr == -1) return -1; /* Invalid character */
        int next = (i + 1 < len) ? roman_val(s[i+1]) : 0;
        if (curr < next) total -= curr;
        else              total += curr;
    }
    return total;
}

1️⃣ Pointers — The Core of C
A pointer stores a memory address. Declare: int *p;. Get address: p = &x;. Dereference: *p = 42;. Null pointer: p = NULL;. Always check for NULL before dereferencing!
2️⃣ Dynamic Memory — malloc & free
malloc(size) allocates memory on the heap. Returns void* — cast it. Always check for NULL return. Always free() when done to avoid memory leaks:
Node *n = (Node*)malloc(sizeof(Node));
if (!n) { /* handle error */ }
free(n);
3️⃣ Struct Self-Reference (Nodes)
A node contains data and a pointer to the next node: typedef struct Node { int data; struct Node *next; } Node;. Note: use struct Node *next inside the struct, not just Node *next (the typedef isn't available yet inside the struct body).
4️⃣ Pointer-to-Pointer (**head)
To modify the head pointer inside a function, pass Node **head. This is a pointer to the pointer. Inside the function: *head = new_node; modifies the caller's head variable. Without this, changes to head don't propagate out.

#include <stdio.h>
#include <stdlib.h>

typedef struct Node { int data; struct Node *next; } Node;

Node *create_node(int data) {
    Node *n = (Node*)malloc(sizeof(Node));
    if (!n) { fprintf(stderr, "malloc failed\n"); exit(1); }
    n->data = data; n->next = NULL;
    return n;
}

void insert_head(Node **head, int data) {
    Node *n = create_node(data);
    n->next = *head;
    *head = n;
}

void insert_tail(Node **head, int data) {
    Node *n = create_node(data);
    if (!*head) { *head = n; return; }
    Node *cur = *head;
    while (cur->next) cur = cur->next;
    cur->next = n;
}

void insert_at(Node **head, int data, int pos) {
    if (pos == 0) { insert_head(head, data); return; }
    Node *cur = *head;
    for (int i = 0; i < pos - 1 && cur; i++) cur = cur->next;
    if (!cur) { printf("Position out of range. Inserting at tail.\n"); insert_tail(head, data); return; }
    Node *n = create_node(data);
    n->next = cur->next;
    cur->next = n;
}

int delete_node(Node **head, int data) {
    if (!*head) return 0;
    if ((*head)->data == data) {
        Node *tmp = *head; *head = (*head)->next; free(tmp); return 1;
    }
    Node *prev = *head, *cur = (*head)->next;
    while (cur) {
        if (cur->data == data) { prev->next = cur->next; free(cur); return 1; }
        prev = cur; cur = cur->next;
    }
    return 0;
}

Node *search(Node *head, int data) {
    while (head) { if (head->data == data) return head; head = head->next; }
    return NULL;
}

void reverse(Node **head) {
    Node *prev = NULL, *curr = *head, *next = NULL;
    while (curr) { next = curr->next; curr->next = prev; prev = curr; curr = next; }
    *head = prev;
}

/* Bubble sort */
void sort(Node *head) {
    if (!head) return;
    int swapped;
    do {
        swapped = 0;
        Node *cur = head;
        while (cur->next) {
            if (cur->data > cur->next->data) {
                int tmp = cur->data; cur->data = cur->next->data; cur->next->data = tmp;
                swapped = 1;
            }
            cur = cur->next;
        }
    } while(swapped);
}

void print_list(Node *head) {
    if (!head) { printf("[empty]\n"); return; }
    printf("HEAD → ");
    while (head) { printf("[%d]", head->data); if (head->next) printf(" → "); head = head->next; }
    printf(" → NULL\n");
}

int length(Node *head) {
    int len = 0; while (head) { len++; head = head->next; } return len;
}

void free_list(Node **head) {
    Node *tmp;
    while (*head) { tmp = *head; *head = (*head)->next; free(tmp); }
}

📚 Mini Course — LIFO, FIFO & Abstract Data Types

1️⃣ Stack (LIFO — Last In, First Out)
Think of a stack of plates. You add to the top (push) and take from the top (pop). Operations: push(value), pop(), peek() (view top without removing), is_empty(). Array implementation: keep a top index.
2️⃣ Queue (FIFO — First In, First Out)
Think of a queue at a ticket counter. Add to the back (enqueue), remove from the front (dequeue). Array implementation uses two pointers: front and rear. Use a circular array to avoid wasting space.
3️⃣ Reverse-Polish Notation (RPN)
In RPN, operators come after operands: "5 3 +" = 5+3 = 8. Algorithm: scan tokens left-to-right. If number → push it. If operator → pop two operands, apply operator, push result. Final answer is the last item on the stack.

  2. Build circular array-based queue with front, rear, size
  3. Write RPN evaluator: tokenize input string by spaces, process each token

  5. Demo menu showing all operations

💡 Hints

1
For circular queue: rear = (rear + 1) % CAPACITY to wrap around. Full condition: size == CAPACITY. Empty: size == 0.
2
To tokenize the RPN string, use strtok(expr, " ") from string.h. Use strtod(token, &end) to check if a token is a number (end != token means it parsed a number).
3
In RPN, pop in order: b = pop(); a = pop(); then compute a OP b. Order matters for subtraction and division!

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

#define CAP 64

/* ── Array-based Stack ── */
typedef struct { double data[CAP]; int top; } Stack;

void stack_init(Stack *s) { s->top = -1; }
int  stack_empty(Stack *s) { return s->top == -1; }
int  stack_full(Stack *s)  { return s->top == CAP - 1; }
void stack_push(Stack *s, double v) {
    if (stack_full(s)) { printf("Stack overflow!\n"); return; }
    s->data[++s->top] = v;
}
double stack_pop(Stack *s) {
    if (stack_empty(s)) { printf("Stack underflow!\n"); return 0; }
    return s->data[s->top--];
}
double stack_peek(Stack *s) { return stack_empty(s) ? 0 : s->data[s->top]; }

/* ── Circular Queue ── */
typedef struct { int data[CAP]; int front, rear, size; } Queue;

void queue_init(Queue *q) { q->front = q->size = 0; q->rear = CAP - 1; }
int  queue_empty(Queue *q) { return q->size == 0; }
int  queue_full(Queue *q)  { return q->size == CAP; }
void enqueue(Queue *q, int v) {
    if (queue_full(q)) { printf("Queue full!\n"); return; }
    q->rear = (q->rear + 1) % CAP;
    q->data[q->rear] = v;
    q->size++;
}
int dequeue(Queue *q) {
    if (queue_empty(q)) { printf("Queue empty!\n"); return -1; }
    int v = q->data[q->front];
    q->front = (q->front + 1) % CAP;
    q->size--;
    return v;
}

/* ── RPN Evaluator ── */
double eval_rpn(const char *expr) {
    Stack s; stack_init(&s);
    char buf[256]; strncpy(buf, expr, 255);
    char *token = strtok(buf, " ");
    while (token) {
        char *end;
        double val = strtod(token, &end);
        if (end != token) {
            /* It's a number */
            stack_push(&s, val);
        } else {
            /* It's an operator */
            if (stack_empty(&s)) { printf("Error: not enough operands\n"); return 0; }
            double b = stack_pop(&s);
            if (stack_empty(&s)) { printf("Error: not enough operands\n"); return 0; }
            double a = stack_pop(&s);
            switch(token[0]) {
                case '+': stack_push(&s, a + b); break;
                case '-': stack_push(&s, a - b); break;
                case '*': stack_push(&s, a * b); break;
                case '/':
                    if (b == 0) { printf("Error: division by zero\n"); return 0; }
                    stack_push(&s, a / b); break;
                default: printf("Unknown operator '%c'\n", token[0]); return 0;
            }
        }
        token = strtok(NULL, " ");
    }
    return stack_pop(&s);
}

📚 Mini Course — Trees, Recursion & Traversals

1️⃣ Binary Search Tree Property
For every node: all values in the left subtree are smaller, all in the right subtree are larger. This property allows O(log n) average-case search, insert, and delete (O(n) worst case for unbalanced trees).
2️⃣ Tree Traversals
Three recursive traversals:
• In-order (left→root→right): produces sorted output
• Pre-order (root→left→right): useful for copying trees
• Post-order (left→right→root): useful for deleting trees
3️⃣ BST Deletion (the tricky part)
Three cases: (1) Leaf node → just delete it. (2) One child → replace node with its child. (3) Two children → find in-order successor (smallest in right subtree), copy its value, then delete the successor.

🗺 Build Plan

  1. Define BSTNode struct with data, left, right pointers
  2. Write recursive insert(root, data) — returns new root
  3. Write recursive search(root, data)
  4. Write inorder(), preorder(), postorder()
  5. Write delete(root, data) handling all 3 cases
  6. Write print_tree(root, "", is_right) for sideways visualization

💡 Hints

1
The insert function pattern: if (root == NULL) return create_node(data); if (data < root->data) root->left = insert(root->left, data); else root->right = insert(root->right, data); return root;
2
For the in-order successor (delete case 3): traverse to root->right then keep going left until there's no more left child. That's the minimum of the right subtree.
3
For sideways tree printing, print right subtree first, then root with prefix, then left subtree — it renders the tree rotated 90° left which is readable.

#include <stdio.h>
#include <stdlib.h>

typedef struct BSTNode { int data; struct BSTNode *left, *right; } BSTNode;

BSTNode *new_node(int d) {
    BSTNode *n = (BSTNode*)malloc(sizeof(BSTNode));
    n->data = d; n->left = n->right = NULL; return n;
}

BSTNode *insert(BSTNode *root, int d) {
    if (!root) return new_node(d);
    if (d < root->data) root->left  = insert(root->left,  d);
    else if (d > root->data) root->right = insert(root->right, d);
    return root; /* duplicates ignored */
}

BSTNode *search(BSTNode *root, int d) {
    if (!root || root->data == d) return root;
    return d < root->data ? search(root->left, d) : search(root->right, d);
}

BSTNode *min_node(BSTNode *n) { while (n->left) n = n->left; return n; }

BSTNode *delete_node(BSTNode *root, int d) {
    if (!root) return NULL;
    if (d < root->data) { root->left  = delete_node(root->left,  d); }
    else if (d > root->data) { root->right = delete_node(root->right, d); }
    else {
        /* Found */
        if (!root->left)  { BSTNode *tmp = root->right; free(root); return tmp; }
        if (!root->right) { BSTNode *tmp = root->left;  free(root); return tmp; }
        /* Two children: replace with in-order successor */
        BSTNode *succ = min_node(root->right);
        root->data = succ->data;
        root->right = delete_node(root->right, succ->data);
    }
    return root;
}

void inorder(BSTNode *root)   { if (!root) return; inorder(root->left); printf("%d ", root->data); inorder(root->right); }
void preorder(BSTNode *root)  { if (!root) return; printf("%d ", root->data); preorder(root->left); preorder(root->right); }
void postorder(BSTNode *root) { if (!root) return; postorder(root->left); postorder(root->right); printf("%d ", root->data); }

/* Print tree sideways */
void print_tree(BSTNode *root, char *prefix, int is_right) {
    if (!root) return;
    char new_prefix[256];
    print_tree(root->right, strcat(strcpy(new_prefix, prefix), is_right ? "│   " : "    "), 1);
    printf("%s%s[%d]\n", prefix, is_right ? "┌── " : "└── ", root->data);
    print_tree(root->left,  strcat(strcpy(new_prefix, prefix), is_right ? "    " : "│   "), 0);
}

void free_tree(BSTNode *root) { if (!root) return; free_tree(root->left); free_tree(root->right); free(root); }

Hash Table (from scratch) Console Intermediate
Implement a hash table with separate chaining for collisions. Support insert, lookup, delete, and dynamic resizing. Visualize bucket distribution.
Compile: gcc hash_table.c -o hash_table

📚 Mini Course — Hash Functions & Collision Resolution

1️⃣ What is Hashing?
A hash function converts a key (e.g. a string) into an array index. The ideal: O(1) average lookup. A simple string hash: sum char ASCII values modulo table size. Better: polynomial rolling hash → hash = (hash * 31 + c) % size for each character c.
2️⃣ Collision Resolution — Separate Chaining
Two keys can hash to the same index (collision). In separate chaining, each bucket is a linked list. When inserting: hash the key, append to the list at that bucket. Lookup: hash the key, search the list. Average O(1) with a good hash and low load factor.
3️⃣ Load Factor & Resizing
Load factor = number of elements / number of buckets. When load factor exceeds ~0.7, performance degrades. Resize: allocate a new larger array, rehash all existing entries into it. Common strategy: double the size and round up to next prime.

🗺 Build Plan

  1. Define Entry struct (key string, value, next pointer)
  2. Define HashTable struct (array of Entry pointers, size, count)
  3. Write djb2 hash function
  4. Implement ht_insert(), ht_get(), ht_delete()
  5. Check load factor after insert; resize if > 0.7
  6. Add ht_print_stats() showing bucket usage

💡 Hints

1
djb2 hash (fast, good distribution): unsigned long hash = 5381; while (*s) hash = ((hash << 5) + hash) + *s++; Then return hash % table_size;
2
For separate chaining, when deleting: traverse the linked list at that bucket keeping a prev pointer. When found, set prev->next = curr->next and free the entry. Handle head deletion specially.
3
During resize, create a new table with 2× capacity, then iterate all buckets, for each entry in each chain, rehash and insert into the new table. Free the old table.
🔓 Solution — hash_table.c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define INIT_SIZE 8

typedef struct Entry { char *key; int value; struct Entry *next; } Entry;
typedef struct { Entry **buckets; int size, count; } HashTable;

unsigned long djb2(const char *s) {
    unsigned long h = 5381;
    while (*s) h = ((h << 5) + h) + (unsigned char)(*s++);
    return h;
}

HashTable *ht_create(int size) {
    HashTable *ht = malloc(sizeof(HashTable));
    ht->size = size; ht->count = 0;
    ht->buckets = calloc(size, sizeof(Entry*));
    return ht;
}

void ht_insert(HashTable **pht, const char *key, int value);

void ht_resize(HashTable **pht) {
    HashTable *old = *pht;
    HashTable *newht = ht_create(old->size * 2);
    for (int i = 0; i < old->size; i++) {
        Entry *e = old->buckets[i];
        while (e) { ht_insert(&newht, e->key, e->value); e = e->next; }
    }
    /* Free old */
    for (int i = 0; i < old->size; i++) {
        Entry *e = old->buckets[i];
        while (e) { Entry *n = e->next; free(e->key); free(e); e = n; }
    }
    free(old->buckets); free(old);
    *pht = newht;
    printf("  [Resized table to %d buckets]\n", newht->size);
}

void ht_insert(HashTable **pht, const char *key, int value) {
    HashTable *ht = *pht;
    if ((double)ht->count / ht->size > 0.7) { ht_resize(pht); ht = *pht; }
    int idx = djb2(key) % ht->size;
    /* Update if key exists */
    for (Entry *e = ht->buckets[idx]; e; e = e->next)
        if (strcmp(e->key, key) == 0) { e->value = value; return; }
    /* New entry */
    Entry *e = malloc(sizeof(Entry));
    e->key = strdup(key); e->value = value;
    e->next = ht->buckets[idx]; ht->buckets[idx] = e;
    ht->count++;
}

int ht_get(HashTable *ht, const char *key, int *out) {
    int idx = djb2(key) % ht->size;
    for (Entry *e = ht->buckets[idx]; e; e = e->next)
        if (strcmp(e->key, key) == 0) { *out = e->value; return 1; }
    return 0;
}

int ht_delete(HashTable *ht, const char *key) {
    int idx = djb2(key) % ht->size;
    Entry *prev = NULL, *e = ht->buckets[idx];
    while (e) {
        if (strcmp(e->key, key) == 0) {
            if (prev) prev->next = e->next; else ht->buckets[idx] = e->next;
            free(e->key); free(e); ht->count--; return 1;
        }
        prev = e; e = e->next;
    }
    return 0;
}

void ht_stats(HashTable *ht) {

📚 Mini Course — Sorting Algorithms & Complexity

1️⃣ Bubble Sort — O(n²)
Repeatedly compare adjacent pairs and swap if out of order. Each pass "bubbles" the largest element to the end. Simple but slow. Optimization: stop early if no swaps occurred in a pass.

Pick a pivot, partition so elements left < pivot < elements right, recurse on both halves. O(n²) worst case (sorted input with bad pivot). Use random pivot or median-of-three to avoid this.

void bubble_sort(int *arr, int n) {
    for (int i = 0; i < n-1; i++)
        for (int j = 0; j < n-i-1; j++) {
            comps++;
            if (arr[j] > arr[j+1]) { swap(&arr[j], &arr[j+1]); draw(arr, n, j, j+1); }
        }
}

        int min_idx = i;
        for (int j = i+1; j < n; j++) { comps++; if (arr[j] < arr[min_idx]) min_idx = j; }
        if (min_idx != i) { swap(&arr[i], &arr[min_idx]); draw(arr, n, i, min_idx); }
    }
}

void insertion_sort(int *arr, int n) {
    for (int i = 1; i < n; i++) {
        int key = arr[i], j = i-1;
        while (j >= 0 && arr[j] > key) {
            comps++; arr[j+1] = arr[j]; j--; draw(arr, n, j+1, i);
        }
        arr[j+1] = key; swaps++;
    }
}

void merge(int *arr, int l, int m, int r) {
    int n1=m-l+1, n2=r-m;
    int *L = malloc(n1*sizeof(int)), *R = malloc(n2*sizeof(int));
    memcpy(L, arr+l, n1*sizeof(int)); memcpy(R, arr+m+1, n2*sizeof(int));
    int i=0,j=0,k=l;
    while (i<n1 && j<n2) { comps++; arr[k++] = (L[i]<=R[j]) ? L[i++] : R[j++]; swaps++; }
    while (i<n1) arr[k++]=L[i++]; while (j<n2) arr[k++]=R[j++];
    free(L); free(R);
}

Min-Heap / Priority Queue Console Intermediate
Build a min-heap with insert, extract-min, and heapify. Use it as a priority queue for a simple task scheduler that processes jobs by priority level.
Compile: gcc min_heap.c -o min_heap

📚 Mini Course — Heaps & Priority Queues

1️⃣ Heap Property
A min-heap is a complete binary tree where every parent is ≤ its children. The minimum element is always at the root (index 0). Stored as an array: parent of i = (i-1)/2. Children of i = 2*i+1 (left) and 2*i+2 (right).
2️⃣ Insert (Sift Up)
Add new element at the end of the array. Then "sift up": compare with parent, swap if smaller than parent, repeat until heap property is restored. O(log n) time.
3️⃣ Extract-Min (Sift Down)
Save root (minimum). Move last element to root position. Reduce heap size. Then "sift down": compare with both children, swap with the smaller child if smaller than current, repeat. O(log n) time.

🗺 Build Plan

  1. Define Heap struct with array, size, capacity
  2. Write sift_up() for insert
  3. Write sift_down() for extract-min
  4. Write heap_insert() and heap_extract_min()
  5. Build priority queue demo: add tasks with priorities, process in order

💡 Hints

1
Parent index of i: (i - 1) / 2. Left child: 2*i + 1. Right child: 2*i + 2. These formulas work because the heap is stored as a 0-indexed array.
2
For sift-down: find the smaller of the two children (min_child). If arr[min_child] < arr[i], swap and continue. If both children are larger, stop.
3
For the task scheduler: struct Task has name and priority. Insert tasks in any order. Extract-min always gives you the highest-priority task (lowest priority number).

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_CAP 64

typedef struct { char name[64]; int priority; } Task;
typedef struct { Task data[MAX_CAP]; int size; } MinHeap;

void heap_init(MinHeap *h) { h->size = 0; }

void swap_tasks(Task *a, Task *b) { Task t = *a; *a = *b; *b = t; }

void sift_up(MinHeap *h, int i) {
    while (i > 0) {
        int p = (i-1)/2;
        if (h->data[p].priority > h->data[i].priority) {
            swap_tasks(&h->data[p], &h->data[i]);
            i = p;
        } else break;
    }
}

void sift_down(MinHeap *h, int i) {
    while (1) {
        int l = 2*i+1, r = 2*i+2, min = i;
        if (l < h->size && h->data[l].priority < h->data[min].priority) min = l;
        if (r < h->size && h->data[r].priority < h->data[min].priority) min = r;
        if (min == i) break;
        swap_tasks(&h->data[i], &h->data[min]);
        i = min;
    }
}

int heap_insert(MinHeap *h, const char *name, int priority) {
    if (h->size >= MAX_CAP) return 0;
    strncpy(h->data[h->size].name, name, 63);
    h->data[h->size].priority = priority;
    sift_up(h, h->size++);
    return 1;
}

int heap_extract_min(MinHeap *h, Task *out) {
    if (h->size == 0) return 0;
    *out = h->data[0];
    h->data[0] = h->data[--h->size];
    if (h->size > 0) sift_down(h, 0);
    return 1;
}

void heap_print(MinHeap *h) {
    printf("  Priority Queue (%d tasks):\n", h->size);
    for (int i = 0; i < h->size; i++)
        printf("    [%d] P=%d: %s\n", i, h->data[i].priority, h->data[i].name);
}

int main() {
    MinHeap h; heap_init(&h);
    int choice; Task t;
    char name[64]; int prio;

    printf("╔══════════════════════════════╗\n");
    printf("║   Priority Queue Scheduler   ║\n");
    printf("╚══════════════════════════════╝\n");

    /* Pre-load tasks */
    heap_insert(&h, "Backup database",   5);
    heap_insert(&h, "Fix critical bug",  1);
    heap_insert(&h, "Code review",       3);
    heap_insert(&h, "Deploy hotfix",     2);
    heap_insert(&h, "Write docs",        7);

    do {
        printf("\n  1) Add task  2) Process next  3) View queue  0) Exit\n  › ");
        scanf("%d",&choice); while(getchar()!='\n');
        switch(choice) {
            case 1:
                printf("  Task name: "); fgets(name,64,stdin); name[strcspn(name,"\n")]='\0';
                printf("  Priority (1=highest): "); scanf("%d",&prio); while(getchar()!='\n');
                heap_insert(&h, name, prio);
                printf("  ✓ Task added.\n"); break;
            case 2:
                if (heap_extract_min(&h, &t))
                    printf("  ⚙ Processing: [P%d] %s\n", t.priority, t.name);
                else printf("  No tasks in queue.\n"); break;
            case 3: heap_print(&h); break;
        }
    } while(choice != 0);

    printf("\n  Remaining tasks (processed in order):\n");
    while (heap_extract_min(&h, &t))
        printf("  [P%d] %s\n", t.priority, t.name);

    return 0;
}

WordFreq freq[MAX_WORDS];
int freq_count = 0;

void lowercase(char *s) { for (; *s; s++) *s = tolower((unsigned char)*s); }

int find_or_add(const char *word) {
    for (int i = 0; i < freq_count; i++) {
        if (strcmp(freq[i].word, word) == 0) { freq[i].count++; return 0; }

        strncpy(freq[freq_count].word, word, MAX_WORD_LEN-1);

        freq_count++;

    return 1;
}

int cmp_freq(const void *a, const void *b) {

}

📚 Mini Course — Processes, fork() & exec()

1️⃣ Process Creation with fork()
fork() creates an exact copy of the current process. It returns: 0 in the child process, the child's PID in the parent, -1 on error. After fork(), you typically call exec() in the child to replace it with a new program.
2️⃣ Executing Programs with exec()
execvp(cmd, args) replaces the current process with a new program. The 'v' means it takes an argv array. The 'p' means it searches PATH. If execvp returns at all, it failed. The parent calls waitpid(pid, &status, 0) to wait for the child.
3️⃣ Pipes
pipe(fd) creates two file descriptors: fd[0] for reading, fd[1] for writing. For cmd1 | cmd2: fork two children, connect cmd1's stdout to the pipe write end, cmd2's stdin to the pipe read end using dup2(fd[1], STDOUT_FILENO).
4️⃣ I/O Redirection
dup2(newfd, oldfd) makes oldfd point to the same file as newfd. To redirect stdout to a file: int fd = open("out.txt", O_WRONLY|O_CREAT|O_TRUNC, 0644); dup2(fd, STDOUT_FILENO); close(fd);

void add_history(const char *cmd) {
    if (hist_count < MAX_HIST) strncpy(history[hist_count++], cmd, MAX_LINE-1);

int parse(char *line, char **args) {
    int i = 0;
    char *tok = strtok(line, " \t\r\n");
    while (tok && i < MAX_ARGS-1) { args[i++] = tok; tok = strtok(NULL, " \t\r\n"); }
    args[i] = NULL;

/* Handle simple command with optional > < redirection */
void execute(char **args) {
    int redir_out = -1, redir_in = -1;
    char *exec_args[MAX_ARGS]; int ea = 0;

    for (int i = 0; args[i]; i++) {
        if (strcmp(args[i], ">") == 0 && args[i+1]) {
            redir_out = open(args[++i], O_WRONLY|O_CREAT|O_TRUNC, 0644);
        } else if (strcmp(args[i], "<") == 0 && args[i+1]) {
            redir_in = open(args[++i], O_RDONLY);
        } else {
            exec_args[ea++] = args[i];
        }
    }
    exec_args[ea] = NULL;

    pid_t pid = fork();
    if (pid == 0) {
        /* Child */
        if (redir_out != -1) { dup2(redir_out, STDOUT_FILENO); close(redir_out); }
        if (redir_in  != -1) { dup2(redir_in,  STDIN_FILENO);  close(redir_in);  }
        execvp(exec_args[0], exec_args);
        fprintf(stderr, "shell: command not found: %s\n", exec_args[0]);
        exit(127);
    } else if (pid > 0) {
        if (redir_out != -1) close(redir_out);
        if (redir_in  != -1) close(redir_in);
        waitpid(pid, NULL, 0);
    } else {
        perror("fork");
    }
}

/* Handle single pipe: cmd1 | cmd2 */
void execute_pipe(char **left, char **right) {
    int pipefd[2];
    if (pipe(pipefd) == -1) { perror("pipe"); return; }

    pid_t p1 = fork();
    if (p1 == 0) {
        dup2(pipefd[1], STDOUT_FILENO);
        close(pipefd[0]); close(pipefd[1]);
        execvp(left[0], left);
        fprintf(stderr, "shell: %s: not found\n", left[0]); exit(127);
    }

}

📚 Mini Course — Complex Numbers & Fractals

1️⃣ The Mandelbrot Set
For each pixel (mapped to a complex number c = x + iy), iterate: z = z² + c, starting from z = 0. If |z| stays ≤ 2 after MAX_ITER iterations, the point is "in the set" (color black). If it escapes, color based on how many iterations it took.
2️⃣ Smooth Coloring
Basic coloring gives harsh bands. Smooth coloring: when |z| > 2, use the fractional escape count: smooth = iter - log2(log2(|z|)). Map this to a color palette using HSV or a cosine palette for beautiful gradients.
3️⃣ Viewport Mapping
Map pixel (px, py) to complex plane: cx = min_x + px * (max_x - min_x) / WIDTH. To zoom: shrink the [min_x, max_x] range around the mouse cursor. Pan: add mouse delta to min/max.

2

#include <SDL2/SDL.h>
#include <stdio.h>
#include <math.h>
#include <stdint.h>

#define W 900
#define H 600
#define MAX_ITER 256

typedef struct { double minx, maxx, miny, maxy; } Viewport;

/* Map pixel to complex plane */
static inline double mapx(int px, Viewport *vp) {
    return vp->minx + px * (vp->maxx - vp->minx) / W;
}
static inline double mapy(int py, Viewport *vp) {
    return vp->miny + py * (vp->maxy - vp->miny) / H;
}

/* Returns smooth escape count, or -1 if in set */
double mandelbrot(double cr, double ci, int max_iter) {
    double zr = 0, zi = 0;
    for (int i = 0; i < max_iter; i++) {
        double zr2 = zr*zr, zi2 = zi*zi;
        if (zr2 + zi2 > 4.0) {
            return i - log2(log2(sqrt(zr2+zi2)));
        }
        zi = 2*zr*zi + ci;
        zr = zr2 - zi2 + cr;
    }
    return -1;
}

double julia(double zr, double zi, double cr, double ci, int max_iter) {
    for (int i = 0; i < max_iter; i++) {
        double zr2 = zr*zr, zi2 = zi*zi;
        if (zr2 + zi2 > 4.0) return i - log2(log2(sqrt(zr2+zi2)));
        double tmp = zr2 - zi2 + cr;
        zi = 2*zr*zi + ci;
        zr = tmp;
    }
    return -1;
}

uint32_t colorize(double t) {
    if (t < 0) return 0xFF000000; /* Black = in set */
    t = fmod(t / MAX_ITER, 1.0);
    uint8_t r = (uint8_t)(255 * (0.5 + 0.5 * cos(6.28318 * (0.00 + t))));
    uint8_t g = (uint8_t)(255 * (0.5 + 0.5 * cos(6.28318 * (0.33 + t))));
    uint8_t b = (uint8_t)(255 * (0.5 + 0.5 * cos(6.28318 * (0.67 + t))));
    return 0xFF000000 | (r << 16) | (g << 8) | b;
}

static uint32_t pixels[W * H];

void render(SDL_Texture *tex, Viewport *vp, int julia_mode, double jcr, double jci) {
    for (int py = 0; py < H; py++) {
        for (int px = 0; px < W; px++) {
            double cx = mapx(px, vp), cy = mapy(py, vp);
            double t = julia_mode ? julia(cx, cy, jcr, jci, MAX_ITER)
                                  : mandelbrot(cx, cy, MAX_ITER);
            pixels[py * W + px] = colorize(t);
        }
    }
    SDL_UpdateTexture(tex, NULL, pixels, W * sizeof(uint32_t));
}

📚 Mini Course — Cellular Automata & Double Buffering

1️⃣ Conway's Rules
For each cell: count its 8 neighbors. (1) Live cell with 2-3 live neighbors → stays alive. (2) Live cell with <2 or >3 neighbors → dies. (3) Dead cell with exactly 3 live neighbors → becomes alive. All updates happen simultaneously.
2️⃣ Double Buffering
Never update the grid in-place! Use two grids: current and next. Read from current, write to next. After processing all cells, swap the pointers. This ensures all cells see the same "snapshot" when computing the next generation.
3️⃣ Wrapping / Toroidal Grid
For neighbor counting at the edges, wrap around: int ny = (y + dy + ROWS) % ROWS;. This creates a "toroidal" topology where the grid wraps like a donut, preventing edge artifacts.

  5. Mouse click: toggle cell at mouse position
  6. Keys: SPACE=pause, R=random, C=clear, +/-=speed

#define ROWS 60
#define W    (COLS * CELL)

uint8_t grid[2][ROWS][COLS];
int cur = 0;

int count_neighbors(int y, int x) {
    int count = 0;
    for (int dy = -1; dy <= 1; dy++)
        for (int dx = -1; dx <= 1; dx++) {
            if (dx == 0 && dy == 0) continue;
            count += grid[cur][(y+dy+ROWS)%ROWS][(x+dx+COLS)%COLS];
        }
    return count;
}

void step() {
    int nxt = 1 - cur;
    for (int y = 0; y < ROWS; y++)
        for (int x = 0; x < COLS; x++) {
            int n = count_neighbors(y, x);
            int alive = grid[cur][y][x];
            grid[nxt][y][x] = (alive && (n==2||n==3)) || (!alive && n==3) ? 1 : 0;
        }
    cur = nxt;
}

void randomize() {
    for (int y = 0; y < ROWS; y++)
        for (int x = 0; x < COLS; x++)
            grid[cur][y][x] = rand() % 4 == 0 ? 1 : 0;
}

📚 Mini Course — Emulation & Bytecode Interpretation

1️⃣ What is Emulation?
An emulator simulates another hardware system in software. CHIP-8 is a virtual machine from the 1970s — perfect first emulation project. It has a fetch-decode-execute cycle: read 2 bytes (opcode) → decode what instruction it is → execute it → repeat.
2️⃣ CHIP-8 Architecture
4096 bytes RAM. 16 general-purpose registers V0-VF. 16-bit index register I. Program counter (PC). Stack (16 levels). Two timers (delay, sound) counting down at 60Hz. 64×32 1-bit display. 16-key hex keypad.
3️⃣ Decoding Opcodes
CHIP-8 opcodes are 2 bytes. Fetch: opcode = (mem[PC] << 8) | mem[PC+1]; PC += 2;. Extract nibbles: type = opcode >> 12, x = (opcode>>8)&0xF, y = (opcode>>4)&0xF, n = opcode&0xF, nn = opcode&0xFF, nnn = opcode&0xFFF.
4️⃣ Display (XOR Drawing)
CHIP-8 sprites are 8px wide, N px tall. The Dxyn opcode XORs sprite data onto the display. If any pixel is turned OFF (1 XOR 1 = 0), set VF = 1 (collision detection). Sprites wrap around screen edges.

  4. Main loop: fetch opcode, decode, execute, update timers at 60Hz

  6. SDL2 render: scale 64×32 display to window size

💡 Hints

#define DISPLAY_W 64
#define DISPLAY_H 32
#define SCALE     12

#define NUM_REGS  16

typedef struct {

void chip8_step(Chip8 *c) {
    uint16_t op = (c->mem[c->PC] << 8) | c->mem[c->PC+1];
    c->PC += 2;

    uint8_t  type = op >> 12;
    uint8_t  x    = (op >> 8) & 0xF;
    uint8_t  y    = (op >> 4) & 0xF;
    uint8_t  n    = op & 0xF;
    uint8_t  nn   = op & 0xFF;
    uint16_t nnn  = op & 0xFFF;

    switch (type) {
    case 0x0:
        if (op == 0x00E0) memset(c->display, 0, sizeof(c->display));
        else if (op == 0x00EE) { c->PC = c->stack[--c->SP]; }
        break;
    case 0x1: c->PC = nnn; break;
    case 0x2: c->stack[c->SP++] = c->PC; c->PC = nnn; break;
    case 0x3: if (c->V[x] == nn) c->PC += 2; break;
    case 0x4: if (c->V[x] != nn) c->PC += 2; break;
    case 0x5: if (c->V[x] == c->V[y]) c->PC += 2; break;
    case 0x6: c->V[x] = nn; break;
    case 0x7: c->V[x] += nn; break;
    case 0x8:
        switch (n) {
        case 0x0: c->V[x] = c->V[y]; break;
        case 0x1: c->V[x] |= c->V[y]; break;
        case 0x2: c->V[x] &= c->V[y]; break;
        case 0x3: c->V[x] ^= c->V[y]; break;
        case 0x4: { uint16_t s=c->V[x]+c->V[y]; c->V[0xF]=s>0xFF?1:0; c->V[x]=s&0xFF; } break;
        case 0x5: c->V[0xF]=c->V[x]>c->V[y]?1:0; c->V[x]-=c->V[y]; break;
        case 0x6: c->V[0xF]=c->V[x]&1; c->V[x]>>=1; break;
        case 0x7: c->V[0xF]=c->V[y]>c->V[x]?1:0; c->V[x]=c->V[y]-c->V[x]; break;
        case 0xE: c->V[0xF]=(c->V[x]>>7)&1; c->V[x]<<=1; break;
        } break;
    case 0x9: if (c->V[x] != c->V[y]) c->PC += 2; break;
    case 0xA: c->I = nnn; break;
    case 0xB: c->PC = nnn + c->V[0]; break;
    case 0xC: c->V[x] = (rand() & 0xFF) & nn; break;
    case 0xD: {
        uint8_t px = c->V[x], py = c->V[y];
        c->V[0xF] = 0;
        for (int row = 0; row < n; row++) {
            uint8_t sprite = c->mem[c->I + row];
            for (int col = 0; col < 8; col++) {
                if (sprite & (0x80 >> col)) {
                    int sx = (px + col) % DISPLAY_W;
                    int sy = (py + row) % DISPLAY_H;
                    if (c->display[sy][sx]) c->V[0xF] = 1;
                    c->display[sy][sx] ^= 1;
                }
            }
        }
        break;
    }
    case 0xE:
        if (nn == 0x9E && c->keys[c->V[x]]) c->PC += 2;
        if (nn == 0xA1 && !c->keys[c->V[x]]) c->PC += 2;
        break;
    case 0xF:
        switch (nn) {
        case 0x07: c->V[x] = c->delay_timer; break;
        case 0x0A: {
            int found = 0;
            for (int k=0;k<16;k++) if (c->keys[k]) { c->V[x]=k; found=1; break; }
            if (!found) c->PC -= 2;
            break;
        }
        case 0x15: c->delay_timer = c->V[x]; break;
        case 0x18: c->sound_timer = c->V[x]; break;
        case 0x1E: c->I += c->V[x]; break;
        case 0x29: c->I = c->V[x] * 5; break;
        case 0x33:
            c->mem[c->I]   = c->V[x] / 100;
            c->mem[c->I+1] = (c->V[x] / 10) % 10;
            c->mem[c->I+2] = c->V[x] % 10;
            break;
        case 0x55: for (int i=0;i<=x;i++) c->mem[c->I+i]=c->V[i]; break;
        case 0x65: for (int i=0;i<=x;i++) c->V[i]=c->mem[c->I+i]; break;
        } break;
    }
}

int main(int argc, char *argv[]) {
    if (argc < 2) { printf("Usage: %s <rom.ch8>\n", argv[0]); return 1; }

    srand(time(NULL));
    Chip8 c; chip8_init(&c);
    if (!chip8_load(&c, argv[1])) return 1;

    SDL_Init(SDL_INIT_VIDEO | SDL_INIT_AUDIO);
    SDL_Window   *win = SDL_CreateWindow("CHIP-8 Emulator", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,
                                          DISPLAY_W*SCALE, DISPLAY_H*SCALE, 0);
    SDL_Renderer *ren = SDL_CreateRenderer(win, -1, SDL_RENDERER_ACCELERATED|SDL_RENDERER_PRESENTVSYNC);
    SDL_Texture  *tex = SDL_CreateTexture(ren, SDL_PIXELFORMAT_ARGB8888, SDL_TEXTUREACCESS_STREAMING,
                                           DISPLAY_W, DISPLAY_H);

    int running = 1;
    SDL_Event ev;
    Uint32 last_timer = SDL_GetTicks();
    uint32_t pixels[DISPLAY_H * DISPLAY_W];

    while (running) {
        /* ~700 CPU cycles per second */
        for (int i = 0; i < 12; i++) chip8_step(&c);

        /* Update timers at 60Hz */
        Uint32 now = SDL_GetTicks();
        if (now - last_timer >= 16) {
            if (c.delay_timer > 0) c.delay_timer--;
            if (c.sound_timer > 0) c.sound_timer--;
            last_timer = now;
        }

        /* Events */
        while (SDL_PollEvent(&ev)) {
            if (ev.type == SDL_QUIT) running = 0;
            if (ev.type == SDL_KEYDOWN && ev.key.keysym.sym == SDLK_ESCAPE) running = 0;
        }
        const Uint8 *kb = SDL_GetKeyboardState(NULL);
        for (int k = 0; k < 16; k++) c.keys[k] = kb[KEY_MAP[k]];

        /* Render */
        for (int y = 0; y < DISPLAY_H; y++)
            for (int x = 0; x < DISPLAY_W; x++)
                pixels[y*DISPLAY_W+x] = c.display[y][x] ? 0xFFAAFFAA : 0xFF111111;

        SDL_UpdateTexture(tex, NULL, pixels, DISPLAY_W * sizeof(uint32_t));
        SDL_RenderClear(ren);
        SDL_RenderCopy(ren, tex, NULL, NULL);
        SDL_RenderPresent(ren);
        SDL_Delay(1);
    }

    SDL_DestroyTexture(tex); SDL_DestroyRenderer(ren);
    SDL_DestroyWindow(win); SDL_Quit();
    return 0;
}
